AWSTemplateFormatVersion: '2010-09-09'
Description: Basic CloudFormation template to launch an EC2 instance

Resources:
  MySecurityGroup:
    Type: AWS::EC2::SecurityGroup
    Properties:
      GroupDescription: Allow SSH access
      SecurityGroupIngress:
        - IpProtocol: tcp
          FromPort: 22
          ToPort: 22
          CidrIp: 0.0.0.0/0

  MyEC2Instance:
    Type: AWS::EC2::Instance
    Properties:
      InstanceType: t2.micro
      ImageId: ami-0c55b159cbfafe1f0  # Amazon Linux 2 (update as needed)
      SecurityGroups:
        - !Ref MySecurityGroup
      Tags:
        - Key: Name
          Value: BasicEC2Instance
